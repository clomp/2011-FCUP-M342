	\item $f=x^5-4x+1, \qquad I=\langle x^3-x^2+x \rangle.$
	\item $f=x^2-4x+4, \qquad I=\langle x^4-6x^2+12x-8, 2x^3-10x^2+16x-8 \rangle.$

\end{enumerate}

\begin{enumerate}

	x - y &=& 1, \\
	3x + 7y -2z &=& 17.
	\end{array}$$
	\item Em $\RR^4$ ou $\CC^4$: 
$$\begin{array}{rcl}
	x_1 + x_2 - x_3 - x_4 &=& 0, \\
	x_1 - x_2 + x_3 &=& 0. \\
	\end{array}$$
	\item Em $\RR^3$ ou $\CC^3$: 
$$\begin{array}{rcl}
	y - x^3 &=& 0,\\

	\end{array}$$
