\sectitle{Ideais}

\begin{enumerate}
\item  Sejam $I$ um ideal de $K[x_1,\ldots, x_n]$ e $f_1, \ldots, f_s \in K[x_1,\ldots, x_n]$. Mostre que $f_1,\ldots, f_s \in I$ se e só se $\langle f_1,\ldots, f_s\rangle \subseteq I$.
Utilize esse facto para mostrar as seguintes identidades em $K[x,y]$:
\begin{enumerate}
  \item $\langle x+y, x-y\rangle = \langle x,y\rangle$
  \item $\langle x+xy, y+xy,x^2,y^2\rangle = \langle x,y\rangle$
  \item $\langle 2x^2+3y^2-11, x^2-y^2-3\rangle = \langle x^2-4,y^2-1\rangle$
\end{enumerate}
\item Mostre que $V(x+xy,y+xy,x^2,y^2)=V(x,y)=\{(0,0)\}$.
\end{enumerate}
